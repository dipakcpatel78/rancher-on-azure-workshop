az group create --name Deepak-Rancher --location southeastasia


az network vnet create --resource-group Deepak-Rancher \
  --name mylab-vnet --address-prefix 10.0.0.0/16 \
  --subnet-name rancher-subnet --subnet-prefix 10.0.0.0/24

az vm create --resource-group Deepak-Rancher \
  --name rancher \
  --admin-username suse \
  --image SUSE:opensuse-leap-15-3:gen1:2021.10.12 \
  --size Standard_B4ms \
  --generate-ssh-keys \
  --public-ip-sku Basic \
  --vnet-name mylab-vnet \
  --subnet rancher-subnet \
  --os-disk-size-gb 50 \
  --verbose 

az vm open-port --resource-group Deepak-rancher --name rancher --port 443 --priority 100
az vm open-port --resource-group Deepak-Rancher --name rancher --port 30000 - 32767 --priority 101


export RANCHER_IP=$(az vm show -d -g Deepak-Rancher -n rancher --query publicIps -o tsv)
ssh -o StrictHostKeyChecking=no suse@$RANCHER_IP


Once Rancher is install, the Rancher URL is saved in the VM path ~/rancher-on-azure-workshop/scripts. The file name is rancher-url.txt.
While working if you incidently close the Brower page for Rancher, you can go back to the VM & path ~/rancher-on-azure-workshop/scripts to locate it. Note if you have change the password while is highly likely, provide your Rancher password to login. The Bootstrap password only works for the first time. 

suse@rancher:~/rancher-on-azure-workshop/scripts> cat rancher-url.txt
